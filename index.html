var Game_Winner = (S) => 
{
let countOfA=0;
let countOfD=0;
for(i=0;i< S.length;i++)
{
if(S[i] ==="A")
{
    countOfA++;
}
else if(S[i]==="D")
{
    countOfD++;
}
}
if(countOfA===countOfD)
{
return "Draw";
}
else if(countOfA > countOfD)
{
return "Aditya";
}
else if(countOfD > countOfA)
{
return "Danish";
}
}
